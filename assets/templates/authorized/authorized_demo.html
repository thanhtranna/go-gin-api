How to enable KEY and SECRET for the caller?

1. Add caller, enter caller ID, caller docking person, remarks and other information;

2. Interfaces that can be called by authorized callers;

3. For details, set the caller’s KEY、SECRET sent to the caller;

How the caller passes the token？

The signature information is stored based on the two parameters Authorization and Authorization-Date in the HTTP Header.

1. Authorization stores signature information, format: caller KEY + space separator + abstract (encrypted string), for example:

Authorization:blog MjJjMDE1MWFkZjMwOWFmYjFlNzViNDFjYjYwMWFlMmM=

2. Authorization-Date stores time information, format: 0000-00-00 00:00:00, use Asia/Shanghai time zone, for example;

Authorization-Date:2021-04-03 21:12:36

func New(key, secret string, ttl time.Duration) Signature {
	return &signature{
		key:    key,
		secret: secret,
		ttl:    ttl,
	}
}

// Generate
// path the requested path (without querystring)
func (s *signature) Generate(path string, method string, params url.Values) (authorization, date string, err error) {
	if path == "" {
		err = errors.New("path required")
		return
	}

	if method == "" {
		err = errors.New("method required")
		return
	}

	methodName := strings.ToUpper(method)
	if !methods[methodName] {
		err = errors.New("method param error")
		return
	}

	// Date
	date = time_parse.CSTLayoutString()

	// Encode() comes in the method sorted by key
	sortParamsEncode, err := url.QueryUnescape(params.Encode())
	if err != nil {
		err = errors.Errorf("url QueryUnescape %v", err)
		return
	}

	// Encrypted string rules
	buffer := bytes.NewBuffer(nil)
	buffer.WriteString(path)
	buffer.WriteString(delimiter)
	buffer.WriteString(methodName)
	buffer.WriteString(delimiter)
	buffer.WriteString(sortParamsEncode)
	buffer.WriteString(delimiter)
	buffer.WriteString(date)

	// Perform sha256 encryption on the data and base64 encode
	hash := hmac.New(sha256.New, []byte(s.secret))
	hash.Write(buffer.Bytes())
	digest := base64.StdEncoding.EncodeToString(hash.Sum(nil))

	authorization = fmt.Sprintf("%s %s", s.key, digest)
	return
}
